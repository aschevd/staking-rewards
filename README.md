# staking-rewards
def staking_rewards(initial_amount, apr, days, compound=False, comp_period=365):
    """
    –†–∞—Å—Å—á–∏—Ç—ã–≤–∞–µ—Ç –¥–æ—Ö–æ–¥–Ω–æ—Å—Ç—å —Å—Ç–µ–π–∫–∏–Ω–≥–∞.
    
    :param initial_amount: –ù–∞—á–∞–ª—å–Ω–∞—è —Å—É–º–º–∞ —Å—Ç–µ–π–∫–∏–Ω–≥–∞ (–≤ —Ç–æ–∫–µ–Ω–∞—Ö)
    :param apr: –ì–æ–¥–æ–≤–∞—è –ø—Ä–æ—Ü–µ–Ω—Ç–Ω–∞—è —Å—Ç–∞–≤–∫–∞ (–≤ %)
    :param days: –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ –¥–Ω–µ–π —Å—Ç–µ–π–∫–∏–Ω–≥–∞
    :param compound: –ï—Å–ª–∏ True, —Å—á–∏—Ç–∞–µ—Ç —Å–ª–æ–∂–Ω—ã–π –ø—Ä–æ—Ü–µ–Ω—Ç (–∫–æ–º–ø–∞—É–Ω–¥–∏–Ω–≥)
    :param comp_period: –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ –¥–Ω–µ–π –≤ –≥–æ–¥—É (–ø–æ —É–º–æ–ª—á–∞–Ω–∏—é 365)
    :return: –ò—Ç–æ–≥–æ–≤–∞—è —Å—É–º–º–∞ + —á–∏—Å—Ç–∞—è –ø—Ä–∏–±—ã–ª—å
    """
    rate = apr / 100  # –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ–º –ø—Ä–æ—Ü–µ–Ω—Ç –≤ –¥–µ—Å—è—Ç–∏—á–Ω—É—é –¥—Ä–æ–±—å
    
    if compound:
        # –§–æ—Ä–º—É–ª–∞ —Å–ª–æ–∂–Ω—ã—Ö –ø—Ä–æ—Ü–µ–Ω—Ç–æ–≤: A = P * (1 + r/n)^(nt)
        final_amount = initial_amount * (1 + rate / comp_period) ** days
    else:
        # –ü—Ä–æ—Å—Ç—ã–µ –ø—Ä–æ—Ü–µ–Ω—Ç—ã: A = P + P * (r * t)
        final_amount = initial_amount + (initial_amount * rate * days / 365)
    
    profit = final_amount - initial_amount
    return final_amount, profit

# –í–≤–æ–¥ –¥–∞–Ω–Ω—ã—Ö –æ—Ç –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
initial = float(input("–í–≤–µ–¥–∏—Ç–µ —Å—É–º–º—É –¥–µ–ø–æ–∑–∏—Ç–∞ (–≤ —Ç–æ–∫–µ–Ω–∞—Ö): "))
apr = float(input("–í–≤–µ–¥–∏—Ç–µ –≥–æ–¥–æ–≤—É—é –ø—Ä–æ—Ü–µ–Ω—Ç–Ω—É—é —Å—Ç–∞–≤–∫—É (APR) –≤ %: "))
days = int(input("–í–≤–µ–¥–∏—Ç–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –¥–Ω–µ–π —Å—Ç–µ–π–∫–∏–Ω–≥–∞: "))
compound_choice = input("–ò—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å —Å–ª–æ–∂–Ω—ã–π –ø—Ä–æ—Ü–µ–Ω—Ç? (y/n): ").strip().lower()

# –í—ã—á–∏—Å–ª–µ–Ω–∏–µ –¥–æ—Ö–æ–¥–Ω–æ—Å—Ç–∏
final_balance, profit = staking_rewards(initial, apr, days, compound=(compound_choice == 'y'))

# –í—ã–≤–æ–¥ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤
print(f"\nüîπ –ò—Ç–æ–≥–æ–≤–∞—è —Å—É–º–º–∞: {final_balance:.6f} —Ç–æ–∫–µ–Ω–æ–≤")
print(f"üí∞ –ß–∏—Å—Ç–∞—è –ø—Ä–∏–±—ã–ª—å: {profit:.6f} —Ç–æ–∫–µ–Ω–æ–≤")
